// src/main/kotlin/api/ApiClient.kt
suspend fun fetchChecklist(): List<ChecklistEstado> {
    val response = window.fetch("/api/v1/checklist").await()
    return response.json().await()
        .unsafeCast<Array<dynamic>>()
        .map { ChecklistEstado(it.id, it.titulo, it.completado, it.fase) }
}

suspend fun toggleItem(id: String) {
    window.fetch("/api/v1/checklist/$id/toggle",
        RequestInit(method = "PATCH")
    ).await()
}
