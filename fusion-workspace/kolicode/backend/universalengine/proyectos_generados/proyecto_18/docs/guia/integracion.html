Manejo de errores

// Patrón recomendado en frontend
suspend fun safeToggle(id: String) {
    try {
        toggleItem(id)
    } catch (e: Exception) {
        console.error("Error al actualizar item $id: ${e.message}")
        // Mostrar feedback visual al usuario
        showError("No se pudo actualizar. Reintenta.")
    }
}
