Lógica de ChecklistEstado.kt

@Serializable
data class ChecklistEstado(
    val id:          String,
    val titulo:      String,
    val completado:  Boolean = false,
    val fase:        String,
    val actualizadoEn: Long = System.currentTimeMillis()
)

// Repositorio en memoria (reemplazar con DB en producción)
object ChecklistRepo {
    private val items = mutableListOf<ChecklistEstado>()
    fun getAll()  = items.toList()
    fun toggle(id: String) {
        val idx = items.indexOfFirst { it.id == id }
        if (idx >= 0) items[idx] = items[idx].copy(
            completado    = !items[idx].completado,
            actualizadoEn = System.currentTimeMillis()
        )
    }
}

Endpoints REST — ChecklistApi.kt

fun Application.checklistRoutes() {
    routing {
        route("/api/v1") {
            get("/checklist") {
                call.respond(ChecklistRepo.getAll())
            }
            patch("/checklist/{id}/toggle") {
                val id = call.parameters["id"]
                    ?: return@patch call.respond(HttpStatusCode.BadRequest)
                ChecklistRepo.toggle(id)
                call.respond(HttpStatusCode.OK)
            }
            get("/webgl/config") {
                call.respond(WebGlConfig(
                    fov = 75.0, background = "#0d0d1a",
                    mesh = "torus", wireframe = true
                ))
            }
        }
    }
}
